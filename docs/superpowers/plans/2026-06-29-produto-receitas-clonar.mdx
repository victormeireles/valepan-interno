# Clonar receitas de outro produto — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Adicionar fluxo **Clonar de…** no modal de receitas do produto, com picker, confirmação com aviso de sobrescrita e dois modos: só receitas (quantidades calculadas para o destino) ou receitas + quantidades (cópia literal).

**Architecture:** Lógica pura em `src/domain/produtos/produto-receitas-clone-utils.ts` (diff + apply). Sugestão de quantidade extraída para helper compartilhado com o modal. UI em `ProdutoReceitasClonePickerModal` + `ProdutoReceitasCloneConfirmDialog`, orquestrados por hook `useProdutoReceitasClone` para manter `ProdutoReceitasConfigModal` abaixo de 500 linhas. Sem nova server action — clone é estado local; persistência continua via actions existentes.

**Tech Stack:** Next.js 15, React 19, TypeScript, Vitest, Tailwind, Material Icons, `resolverQuantidadeReceitaParaProduto`.

**Spec:** `docs/superpowers/specs/2026-06-29-produto-receitas-clonar-design.mdx`

```typescript
import type { ProdutoPesoInput } from '@/domain/assadeiras/produto-peso';
import { resolverQuantidadeReceitaParaProduto } from '@/domain/receitas/receita-produto-quantidade-resolver';
import type { ReceitaGramatura } from '@/domain/receitas/receita-gramatura-resolver';
import type { TipoReceita } from '@/components/ProdutosConfig/produto-receita-tipo-options';

export type ReceitaSugestaoInput = {
  id: string;
  ingredientes?: Array<{ quantidade: number; unidade: string | null }>;
  gramaturas?: ReceitaGramatura[];
};

export function buildSugestaoReceitaProduto(
  receita: ReceitaSugestaoInput | undefined,
  tipo: TipoReceita,
  produto: ProdutoPesoInput,
) {
  if (!receita) {
    return { pesoGramas: null, quantidade: null, resumo: null, aviso: null };
  }

  return resolverQuantidadeReceitaParaProduto({
    tipo,
    ingredientes: receita.ingredientes,
    gramaturas: receita.gramaturas,
    produto,
  });
}
```

```typescript
// src/domain/produtos/produto-receitas-clone-utils.ts
import { PRODUTO_RECEITA_TIPO_OPTIONS, type TipoReceita } from '@/components/ProdutosConfig/produto-receita-tipo-options';
import { countReceitasVinculadas } from '@/domain/produtos/produto-config-resumo';
import {
  buildSugestaoReceitaProduto,
  type ReceitaSugestaoInput,
} from '@/domain/receitas/receita-produto-quantidade-sugestao';
import type { ProdutoPesoInput } from '@/domain/assadeiras/produto-peso';
import type {
  CloneApplyResult,
  CloneDestinoFormState,
  CloneDiffLinha,
  CloneModo,
  ProdutoCloneFonte,
} from './produto-receitas-clone-types';

type ApplyContext = {
  receitasCatalogo: ReceitaSugestaoInput[];
  produtoDestino: ProdutoPesoInput;
  destinoAtual?: CloneDestinoFormState;
};

export function filterProdutosFonteClone(
  produtos: ProdutoCloneFonte[],
  produtoAtualId: string,
): ProdutoCloneFonte[] {
  return produtos.filter(
    (item) => item.id !== produtoAtualId && countReceitasVinculadas(item.receitasVinculadas) > 0,
  );
}

export function normalizeCloneSearch(value: string): string {
  return value
    .normalize('NFD')
    .replace(/[\u0300-\u036f]/g, '')
    .toLowerCase()
    .trim();
}

export function matchesCloneSearch(produto: ProdutoCloneFonte, term: string): boolean {
  if (!term) return true;
  const haystack = normalizeCloneSearch(`${produto.nome} ${produto.codigo}`);
  return haystack.includes(normalizeCloneSearch(term));
}

export function buildCloneDiff(
  origem: ProdutoCloneFonte,
  destino: CloneDestinoFormState,
  receitaNomePorId?: Map<string, string>,
): CloneDiffLinha[] {
  const linhas: CloneDiffLinha[] = [];

  for (const option of PRODUTO_RECEITA_TIPO_OPTIONS) {
    const vinculo = origem.receitasVinculadas[option.value];
    if (!vinculo?.ativo) continue;

    const destinoReceitaId = destino.selectedReceitas[option.value];
    const acao: CloneDiffLinha['acao'] = destinoReceitaId ? 'substituir' : 'adicionar';

    linhas.push({
      tipo: option.value,
      tipoLabel: option.label,
      acao,
      destinoReceitaNome: destinoReceitaId
        ? receitaNomePorId?.get(destinoReceitaId)
        : undefined,
      destinoReceitaId: destinoReceitaId || undefined,
      destinoQuantidade: destino.quantidades[option.value],
      origemReceitaNome: vinculo.receita_nome,
      origemReceitaId: vinculo.receita_id,
      origemQuantidade: vinculo.quantidade,
    });
  }

  return linhas;
}

export function hasCloneRecipeChanges(diff: CloneDiffLinha[]): boolean {
  return diff.some(
    (linha) =>
      linha.acao === 'adicionar' || linha.destinoReceitaId !== linha.origemReceitaId,
  );
}

export function applyCloneToFormState(
  origem: ProdutoCloneFonte,
  modo: CloneModo,
  context: ApplyContext,
): CloneApplyResult {
  const selectedReceitas: Partial<Record<TipoReceita, string>> = {
    ...context.destinoAtual?.selectedReceitas,
  };
  const quantidades: Partial<Record<TipoReceita, number>> = {
    ...context.destinoAtual?.quantidades,
  };
  const quantidadesManuais: Partial<Record<TipoReceita, boolean>> = {};

  for (const option of PRODUTO_RECEITA_TIPO_OPTIONS) {
    const vinculo = origem.receitasVinculadas[option.value];
    if (!vinculo?.ativo) continue;

    selectedReceitas[option.value] = vinculo.receita_id;

    if (modo === 'receitas-e-quantidades') {
      quantidades[option.value] = vinculo.quantidade;
      quantidadesManuais[option.value] = true;
      continue;
    }

    const receita = context.receitasCatalogo.find((item) => item.id === vinculo.receita_id);
    const sugestao = buildSugestaoReceitaProduto(receita, option.value, context.produtoDestino);
    quantidades[option.value] = sugestao.quantidade ?? undefined;
    quantidadesManuais[option.value] = false;
  }

  return { selectedReceitas, quantidades, quantidadesManuais };
}
```

```tsx
'use client';

import { useMemo, useState } from 'react';
import type { ProdutoCloneFonte } from '@/domain/produtos/produto-receitas-clone-types';
import { countReceitasVinculadas } from '@/domain/produtos/produto-config-resumo';
import { matchesCloneSearch } from '@/domain/produtos/produto-receitas-clone-utils';

type Props = {
  isOpen: boolean;
  produtoDestinoNome: string;
  produtos: ProdutoCloneFonte[];
  onClose: () => void;
  onSelect: (produto: ProdutoCloneFonte) => void;
};

export default function ProdutoReceitasClonePickerModal({
  isOpen,
  produtoDestinoNome,
  produtos,
  onClose,
  onSelect,
}: Props) {
  const [search, setSearch] = useState('');

  const filtered = useMemo(
    () => produtos.filter((p) => matchesCloneSearch(p, search)),
    [produtos, search],
  );

  if (!isOpen) return null;

  return (
    <div className="fixed inset-0 z-[60] flex items-end md:items-center justify-center p-0 md:p-4">
      <div className="absolute inset-0 bg-black/50 backdrop-blur-sm" onClick={onClose} aria-hidden="true" />
      <div
        role="dialog"
        aria-modal="true"
        aria-labelledby="clone-picker-title"
        className="relative bg-white w-full md:max-w-lg md:rounded-3xl rounded-t-3xl shadow-2xl max-h-[85dvh] flex flex-col"
      >
        <div className="px-5 py-4 border-b border-gray-100">
          <p id="clone-picker-title" className="text-lg font-bold text-gray-900">
            Clonar receitas
          </p>
          <p className="text-sm text-gray-500 mt-1">
            Escolha o produto de origem para <span className="font-medium">{produtoDestinoNome}</span>.
          </p>
          <input
            type="search"
            value={search}
            onChange={(e) => setSearch(e.target.value)}
            placeholder="Buscar por nome ou código…"
            className="mt-3 w-full min-h-11 px-3 py-2 bg-gray-50 border border-gray-200 rounded-xl text-sm"
            autoFocus
          />
        </div>
        <ul className="overflow-y-auto flex-1 divide-y divide-gray-100">
          {filtered.length === 0 ? (
            <li className="px-5 py-8 text-sm text-gray-500 text-center">
              Nenhum produto com receitas encontrado.
            </li>
          ) : (
            filtered.map((produto) => {
              const count = countReceitasVinculadas(produto.receitasVinculadas);
              return (
                <li key={produto.id}>
                  <button
                    type="button"
                    onClick={() => onSelect(produto)}
                    className="w-full min-h-11 px-5 py-3 flex items-center justify-between gap-3 text-left hover:bg-amber-50 transition-colors"
                  >
                    <span className="font-medium text-gray-900 truncate">{produto.nome}</span>
                    <span className="shrink-0 text-xs font-semibold px-2 py-1 rounded-full bg-gray-100 text-gray-700">
                      {count} {count === 1 ? 'receita' : 'receitas'}
                    </span>
                  </button>
                </li>
              );
            })
          )}
        </ul>
        <div className="px-5 py-3 border-t border-gray-100 flex justify-end">
          <button
            type="button"
            onClick={onClose}
            className="min-h-11 px-4 py-2.5 text-sm font-semibold text-gray-700 bg-white border border-gray-200 rounded-xl hover:bg-gray-50"
          >
            Cancelar
          </button>
        </div>
      </div>
    </div>
  );
}
```
